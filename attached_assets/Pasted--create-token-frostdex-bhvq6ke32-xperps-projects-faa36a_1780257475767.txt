### ✅ ثالثاً: إنشاء Hook جديد `src/hooks/useTokenCreation.ts`
هذا هو أهم جزء. لازم يخدم هكا:
```typescript
import { useWallet } from '@solana/wallet-adapter-react';
import { Program, AnchorProvider } from '@coral-xyz/anchor';
import { uploadImage, uploadMetadata } from '../services/ipfs';
import { PROGRAM_ID, ADMIN_WALLET } from '../config/constants';

export function useTokenCreation() {
  const wallet = useWallet();
  const [loading, setLoading] = useState(false);
  const [program, setProgram] = useState<Program | null>(null);

  useEffect(() => {
    if (wallet.publicKey) {
      const provider = new AnchorProvider(connection, wallet, {});
      const program = new Program(IDL, PROGRAM_ID, provider);
      setProgram(program);
    }
  }, [wallet]);

  const createToken = async (formData: any) => {
    if (!wallet.publicKey) throw new Error("Connect wallet");

    // 1. رفع الصورة و الميتاداتا لـ IPFS
    const imageUrl = await uploadImage(formData.imageFile);
    const metadataUri = await uploadMetadata({
      name: formData.name,
      symbol: formData.symbol,
      description: formData.description,
      image: imageUrl,
      social: { ...formData.socialLinks }
    });

    // 2. حساب الرسوم الإجمالية للخيارات المتقدمة
    const totalAdvancedFee = Object.entries(formData.advancedOptions)
      .filter(([, v]) => v)
      .reduce((sum, [key]) => sum + ADVANCED_FEES[key], 0);

    // 3. نادي العقد باش ينشئ التوكن
    const tx = await program.methods
      .createToken(
        formData.name,
        formData.symbol,
        metadataUri,
        totalAdvancedFee, // المبلغ المراد دفعه مقابل الخيارات المتقدمة
        formData.initialBuyEnabled // هل نفعّل Initial Buy؟
      )
      .accounts({
        creator: wallet.publicKey,
        feeRecipient: new PublicKey(ADMIN_WALLET)
      })
      .rpc();

    // 4. إذا كان Initial Buy مفعّل، ننفذ العملية
    if (formData.initialBuyEnabled && parseFloat(formData.initialBuyAmount) > 0) {
      const buyAmount = parseFloat(formData.initialBuyAmount) * 1e9; // lamports
      await program.methods
        .initialBuy(new anchor.BN(buyAmount))
        .accounts({
          user: wallet.publicKey,
          tokenMint: tx, // عنوان التوكن الناتج
          feeRecipient: new PublicKey(ADMIN_WALLET)
        })
        .rpc();
    }

    return { success: true, mintAddress: tx };
  };

  return { createToken, loading };
}
```
